178  void translateTexture(const float x, const float y, const size_t textureID)
179  {
180  for (size_t cn=0; cn<4; ++cn)
181  {
182  osg::Vec2f &currentCoord=(*(texCoordVec_[textureID]))[cn];
183 
184  currentCoord-=osg::Vec2f(x, y);
185  }
186 
187  Geom_->setTexCoordArray(textureID, texCoordVec_[textureID].get());
188  }

190  void scaleTexture(const float cx, const float cy,
191  const float sx, const float sy,
192  const size_t textureID)
193  {
194  for (size_t cn=0; cn<4; ++cn)
195  {
196  osg::Vec2f &currentCoord=(*(texCoordVec_[textureID]))[cn];
197 
198  currentCoord-=osg::Vec2f(cx, cy);
199  currentCoord=osg::componentDivide(currentCoord, osg::Vec2f(sx, sy));
200  currentCoord+=osg::Vec2f(cx, cy);
201  }
202 
203  Geom_->setTexCoordArray(textureID, texCoordVec_[textureID].get());
204  }

206  void rotateTexture(const float cx, const float cy,
207  const float thetaDeg,
208  const size_t textureID)
209  {
210  const float thetaRad=-(thetaDeg/180.0) * M_PI;
211  const osg::Vec2f A(cosf(thetaRad), sinf(thetaRad));
212  const osg::Vec2f B(-A._v[1], A._v[0]);
213 
214  for (size_t cn=0; cn<4; ++cn)
215  {
216  osg::Vec2f &currentCoord=(*(texCoordVec_[textureID]))[cn];
217 
218  currentCoord-=osg::Vec2f(cx, cy);
219  currentCoord=A * (currentCoord._v[0]) + B * (currentCoord._v[1]);
220  currentCoord+=osg::Vec2f(cx, cy);
221  }
222 
223  Geom_->setTexCoordArray(textureID, texCoordVec_[textureID].get());
224  }
