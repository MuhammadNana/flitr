153  if ((startY+i_lY)<0) startY=-i_lY;

313  offsetX=0;
314  outOfBoundsX=true;
315  }

317  {
318  offsetX=(width-1);

320  }
321

324  offsetY=0;

326  }

328  {
329  offsetY=(height-1);

331  }

333 
334 
335  iData+=((offsetX+offsetY*width));
336  const double dA=iData[0];
337  const double dB=(outOfBoundsX) ? dA : iData[(1)];
338 
339  const double dC=(outOfBoundsY) ? dA : iData[(0+width)];
340  const double dD=(outOfBoundsY) ? dB : ((outOfBoundsX) ? dC : iData[(1+width)]);
341
