90 typedef enum _VIDEO_INPUT_FLAGS // can be combined, i.e. (WDM_SHOW_DIALOG|WDM_MATCH_FILTER_NAME)
91 {
92  WDM_MATCH_FORMAT = 0x0001, // first media type matching the a given set of format requirements
93  WDM_SHOW_FORMAT_DIALOG = 0x0002, // displays either the capture pin's property pages (non-DV cameras) or
94  // the DV-decoder filter's format dialog before connecting anything
95  WDM_SHOW_CONTROL_DIALOG = 0x0004, // displays the source filter's property pages before connecting anything
96  WDM_MATCH_FILTER_NAME = 0x0008, // tries find a match based on a filter name substring, i.e. "QuickCam" or "Sony"
97  WDM_MATCH_IEEE1394_ID = 0x0010, // match filter based on a unique 64-bit IEEE 1394 device ID (MSDV driver only)
98  // this technique allows to explicitly choose between different cameras connected
99  // to the same IEEE 1394 bus. Setting this flag usually requires use of the WDM
100  // MSDV driver(qdv.dll)
101  // >> use "/bin/IEEE1394_id.exe" to determine your camera's ID.
102  WDM_USE_MAX_FRAMERATE = 0x0020 // uses the max. available frame rate for the format requested
103 
104 } VIDEO_INPUT_FLAGS;

118 typedef enum _ASYNC_INPUT_FLAGS
119 {
120  ASYNC_INPUT_DO_NOT_USE_CLOCK = 0x0100, // call IMediaFilter::SetSyncSource(NULL) on async file source filter
121  // this will effectively prevent samples from being dropped, but
122  // may result in 100% CPU usage on single-processor systems
123  ASYNC_LOOP_VIDEO = 0x0200, // continuously loops through an input file
124  ASYNC_RENDER_SECONDARY_STREAMS = 0x0400 // The Async File Source Filter is always used in combination with an
125  // AVI Splitter Filter (CLSID_AviSplitter), where Stream 0 is assumed
126  // to contain video data (MEDIATYPE_Video). This flag instructs DSVL_GraphManager
127  // to call IGraphBuilder->Render(IPin*) on Streams 1..n
128  // Set this flag if your AVI file contains an audio stream AND you want it
129  // rendered by DirectSound (usually through CLSID_DSoundRender).
130  // DO NOT SET this flag if your AVI file contains more than one video stream
