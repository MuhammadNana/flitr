337  // Get a character, while interpreting entities.
338  // The length can be from 0 to 4 bytes.
339  inline static const char* GetChar( const char* p, char* _value, int* length, TiXmlEncoding encoding )
340  {
341  assert( p );
342  if ( encoding == TIXML_ENCODING_UTF8 )
343  {
344  *length = utf8ByteTable[ *((const unsigned char*)p) ];
345  assert( *length >= 0 && *length < 5 );
346  }
347  else
348  {
349  *length = 1;
350  }
351 
352  if ( *length == 1 )
353  {
354  if ( *p == '&' )
355  return GetEntity( p, _value, length, encoding );
356  *_value = *p;
357  return p+1;
358  }
359  else if ( *length )
360  {
361  //strncpy( _value, p, *length ); // lots of compilers don't like this function (unsafe),
362  // and the null terminator isn't needed
363  for( int i=0; p[i] && i<*length; ++i ) {
364  _value[i] = p[i];
365  }
366  return p + (*length);
367  }
368  else
369  {
370  // Not valid text.
371  return 0;
372  }
373  }
