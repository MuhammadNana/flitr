58  void four1(float * const data, uint32_t nn, int32_t isign)
59  {
60  unsigned long n,mmax,m,j,istep,i;
61  double wtemp,wr,wpr,wpi,wi,theta;
62  float tempr,tempi;
63 
64  n=nn << 1;
65  j=1;
66 
67  for (i=1;i<n;i+=2)
68  {
69  if (j > i)
70  {
71  std::swap(data[j],data[i]);
72  std::swap(data[j+1],data[i+1]);
73  }
74  m=nn;
75  while (m >= 2 && j > m)
76  {
77  j -= m;
78  m >>= 1;
79  }
80 
81  j += m;
82  }
83 
84  mmax=2;
85 
86  while (n > mmax)
87  {
88  istep=mmax << 1; theta=isign*(6.28318530717959/mmax); wtemp=sin(0.5*theta);
89  wpr = -2.0*wtemp*wtemp;
90 
91  wpi=sin(theta);
92  wr=1.0;
93  wi=0.0;
94 
95  for (m=1;m<mmax;m+=2)
96  {
97  for (i=m;i<=n;i+=istep)
98  {
99  j=i+mmax;
100  tempr=wr*data[j]-wi*data[j+1];
101  tempi=wr*data[j+1]+wi*data[j];
102  data[j]=data[i]-tempr;
103  data[j+1]=data[i+1]-tempi;
104  data[i] += tempr;
105  data[i+1] += tempi;
106  }
107 
108  wr=(wtemp=wr)*wpr-wi*wpi+wr;
109  wi=wi*wpr+wtemp*wpi+wi;
110  }
111 
112  mmax=istep;
113  }
