74  void update(float dT)
75  {
76  if (interval_ == 0.0f)
77  {
78  px_+=dx_ * v_;
79  py_+=dy_ * v_;
80  }
81  else
82  {
83  elapsedTime_ += dT;
84  int index = (int)(elapsedTime_ / interval_);
85 
86  if (index >= (int)samples_.size() - 1)
87  {
88  px_ = samples_.back().first;
89  py_ = samples_.back().second;
90  }
91  else
92  {
93  std::pair<float, float> curSample = samples_[index];
94  std::pair<float, float> nextSample = samples_[index+1];
95  float weight = (elapsedTime_ - (index * interval_)) / interval_;
96 
97  // lerp
98  float x = curSample.first + (nextSample.first - curSample.first) * weight;
99  float y = curSample.second + (nextSample.second - curSample.second) * weight;
100 
101  dx_ = (x - px_);
102  dy_ = (y - py_);
103  //Normalise the direction vector.
104  dx_/=sqrt(dx_*dx_+dy_*dy_);
105  dy_/=sqrt(dx_*dx_+dy_*dy_);
106 
107  px_ = x;
108  py_ = y;
109  }
110 
111  }
112 
113  sa_+=dsa_;
114  sb_+=dsb_;
115  }

122  float getSupportDensity(const float x, const float y) const
123  {
124  float supportDensity=0.0f;
125  unsigned long numSubpixelSamples=0;
126 
127  const float subPixelSize=0.1f;//0.1f, 0.2f, 0.4f
128 
129  const float rx=x-px_;
130  const float ry=y-py_;
131 
132  if ((fabsf(rx)<sa_*4.0f)&&(fabsf(ry)<sa_*4.0f)) //Assumes that sa_>sb_
133  {
134 
135  for (float subPixelY=-(0.5f-subPixelSize*0.5f); subPixelY<0.5f; subPixelY+=subPixelSize)
136  {
137  const float aY=(ry+subPixelY)*dy_;
138  const float bY=-(ry+subPixelY)*dx_;
139 
140  for (float subPixelX=-(0.5f-subPixelSize*0.5f); subPixelX<0.5f; subPixelX+=subPixelSize)
141  {
142  float a=(rx+subPixelX)*dx_ + aY;
143  float b=(rx+subPixelX)*dy_ + bY;
144 
145  supportDensity+=UNGaussian1D(a,sa_)*UNGaussian1D(b,sb_);
146  numSubpixelSamples++;
147  }
148  }
149 
150 
151  return supportDensity/numSubpixelSamples;
152  }
153 
154  return 0.0f;
155  }
