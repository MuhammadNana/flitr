97  virtual void operator () (osg::RenderInfo& renderInfo) const
98  {
99  if (producer_->enabled_)
100  {
101  std::vector<Image**> imv;
102  do {
103  imv = producer_->reserveWriteSlot();
104 
105  if (imv.size()!=1)
106  {
107  if (blocking_)
108  {//sleep and try again i.e. block viewer's frame.
109  OpenThreads::Thread::microSleep(1000);
110  } else
111  {//drop this frame i.e. do not block viewer's frame.
112  logMessage(LOG_CRITICAL) << "Dropping frames - no space in texture capture buffer\n";
113  logMessage(LOG_CRITICAL).flush();
114  return;
115  }
116  }
117  } while (imv.size() != 1);
118 
119 
120  Image* im = *(imv[0]);
121  osg::TextureRectangle* trect_ = producer_->OutputTexture_.get();
122 
123  renderInfo.getState()->applyTextureAttribute(0, //unit
124  trect_);
125 
126  GLenum pixelFormat=GL_RGB;
127  GLenum pixelType=GL_UNSIGNED_BYTE;
128 
129  switch (producer_->OutputFormat_.getPixelFormat())
130  {
131  case ImageFormat::FLITR_PIX_FMT_Y_8:
132  pixelFormat=GL_LUMINANCE;
133  pixelType=GL_UNSIGNED_BYTE;
134  break;
135  case ImageFormat::FLITR_PIX_FMT_RGB_8:
136  pixelFormat=GL_RGB;
137  pixelType=GL_UNSIGNED_BYTE;
138  break;
139  case ImageFormat::FLITR_PIX_FMT_Y_16:
140  pixelFormat=GL_LUMINANCE;
141  pixelType=GL_UNSIGNED_SHORT;
142  break;
143  case ImageFormat::FLITR_PIX_FMT_Y_F32:
144  pixelFormat=GL_RED;
145  pixelType=GL_FLOAT;
146  break;
147  default:
148  std::cerr << "Pixel format " << producer_->OutputFormat_.getPixelFormat() << " not yet implemented in TextureCaptureProducer.\n";
149  exit(-1);
150  }
151 
152  glGetTexImage(trect_->getTextureTarget(), 0, //level
153  pixelFormat, pixelType, im->data());
154 
155  // fill other vars
156  if (producer_->CreateMetadataFunction_)
157  {
158  im->setMetadata(producer_->CreateMetadataFunction_());
159 
160  //if (im->metadata())
161  //{
162  //std::cout << " TextureCaptureProducer frame: " << im->metadata()->getString();
163  //std::cout.flush();
164  //}
165  } else
166  {
167  std::shared_ptr<DefaultTextureCaptureMetadata> meta(new DefaultTextureCaptureMetadata());
168  meta->PCTimeStamp_ = currentTimeNanoSec();
169  im->setMetadata(meta);
170  }
171 
172 
173 
174  if (producer_->SaveNextImageTo_!="")
175  {
176  osg::ref_ptr<osg::Image> image=osg::ref_ptr<osg::Image>(new osg::Image());
177  image->setImage(im->format()->getWidth(), im->format()->getHeight(), 1,
178  pixelFormat, pixelFormat, pixelType, im->data(), osg::Image::NO_DELETE);
179 
180  osg::ref_ptr<osg::Image> imageFlipped=osg::ref_ptr<osg::Image>(new osg::Image(*image,osg::CopyOp::DEEP_COPY_ALL));
181  imageFlipped->flipVertical();
182 
183  osgDB::writeImageFile(*imageFlipped, producer_->SaveNextImageTo_);
184  producer_->SaveNextImageTo_="";
185  }
186 
187  producer_->releaseWriteSlot();
188  }
189  }
190  TextureCaptureProducer* producer_;
191  bool blocking_;//blocking the viewer frame method as opposed to dropping image frames.
192  };
