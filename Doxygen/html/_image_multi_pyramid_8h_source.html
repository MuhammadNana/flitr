316 
317

320  {
321  offsetX=0;
322  outOfBoundsX=true;
323  }
324  if (offsetX>=((long)width-1))
325  {
326  offsetX=(width-1);
327  outOfBoundsX=true;
328  }
329 
330  if (offsetY<0)
331  {
332  offsetY=0;
333  outOfBoundsY=true;
334  }

336  {
337  offsetY=(height-1);
338  outOfBoundsY=true;
339  }
340  //==========================================================================================================//
341 
342 
343  iData+=((offsetX+offsetY*width));
344  const double dA=iData[0];
345  const double dB=(outOfBoundsX) ? dA : iData[(1)];
346 
347  const double dC=(outOfBoundsY) ? dA : iData[(0+width)];
348  const double dD=(outOfBoundsY) ? dB : ((outOfBoundsX) ? dC : iData[(1+width)]);
